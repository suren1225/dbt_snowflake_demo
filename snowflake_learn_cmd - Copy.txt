CREATE OR REPLACE PROCEDURE SALES_DB.CUSTOM_AUDIT.LOAD_AUDIT()
RETURNS STRING
LANGUAGE SQL
AS
$$
BEGIN

    INSERT INTO SALES_DB.CUSTOM_AUDIT.CUSTOMER_AUDIT
    SELECT 'SALES_DB.raw.CUSTOMER','LOAD_AUDIT',count(1),CURRENT_TIMESTAMP()
    FROM SALES_DB.raw.CUSTOMER;
    
    RETURN 'LOAD COMPLETED';

END;
$$;

CREATE OR REPLACE PROCEDURE GET_CUSTOMER_COUNT()
RETURNS STRING
LANGUAGE PYTHON
RUNTIME_VERSION = '3.11'
PACKAGES = ('snowflake-snowpark-python')
HANDLER = 'main'
AS
$$
from snowflake.snowpark import Session

def main(session: Session):

    cnt = session.sql(
        "SELECT COUNT(*) FROM CUSTOMER"

--Window Functions

select current_schema();

use schema raw;

--ROW_NUMBER
SELECT *,
    ROW_NUMBER() OVER(PARTITION BY CUSTOMER_ID ORDER BY ORDER_DATE DESC) RN
FROM ORDERS;

--RANK
SELECT *
    ,RANK() OVER(ORDER BY QUANTITY DESC) 
FROM ORDERS; 

--DENSE_RANK
SELECT *
    , DENSE_RANK() OVER(ORDER BY QUANTITY DESC)
FROM ORDERS;

--LAG
SELECT ORDER_DATE,QUANTITY
    ,LAG(QUANTITY) OVER(ORDER BY ORDER_DATE) PREV_QTY
FROM ORDERS;

--LEAD
SELECT ORDER_DATE, QUANTITY
    , LEAD(QUANTITY) OVER(ORDER BY ORDER_DATE) NEXT_QTY
FROM ORDERS;

--FIRST_VALUE
SELECT CUSTOMER_ID
    , FIRST_VALUE(ORDER_DATE) OVER(PARTITION BY CUSTOMER_ID ORDER BY ORDER_DATE)
FROM ORDERS;
 
--LAST_VALUE
SELECT CUSTOMER_ID,
    LAST_VALUE(ORDER_DATE) 
    OVER( PARTITION BY CUSTOMER_ID
    ORDER BY ORDER_DATE
    ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
)
FROM ORDERS;

--Running Total
SELECT ORDER_DATE, QUANTITY,
    SUM(QUANTITY) OVER(ORDER BY ORDER_DATE) RUNNING_TOTAL
FROM ORDERS;

--Moving Average
SELECT ORDER_DATE,AVG(QUANTITY)
    OVER(
        ORDER BY ORDER_DATE
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) as mon_avg
FROM ORDERS;

--Recursive CTE

CREATE TABLE EMPLOYEE
(
    EMP_ID NUMBER,
    EMP_NAME STRING,
    MANAGER_ID NUMBER
);

show tables;

WITH RECURSIVE EMP_TREE AS
(
    SELECT EMP_ID,EMP_NAME,MANAGER_ID,1 LEVEL
    FROM EMPLOYEE
    WHERE MANAGER_ID IS NULL
    
    UNION ALL
    
    SELECT E.EMP_ID,E.EMP_NAME,E.MANAGER_ID,T.LEVEL+1
    FROM EMPLOYEE E
    JOIN EMP_TREE T ON E.MANAGER_ID=T.EMP_ID
)
SELECT *
FROM EMP_TREE;
